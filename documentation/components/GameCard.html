import { Component, Input, Output, EventEmitter } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterLink } from '@angular/router';
import { Producto } from '../../models/models';
import { ClpPipe } from '../../pipes/clp-pipe';

/**
 * Ficha de un juego dentro del catalogo. Recibe el producto desde el
 * componente padre (Categoria) por @Input y avisa al padre por @Output
 * cuando el usuario quiere agregarlo al carrito.
 */
@Component({
  selector: 'app-game-card',
  imports: [CommonModule, RouterLink, ClpPipe],
  templateUrl: './game-card.html',
})
export class GameCard {
  /** Producto que se muestra en la ficha. */
  @Input({ required: true }) producto!: Producto;

  /** Mensaje de feedback de esta ficha; lo controla el componente padre. */
  @Input() feedback: { texto: string; tipo: 'ok' | 'err' } | null = null;

  /** Se emite cuando el usuario presiona "Agregar al carrito". */
  @Output() agregar = new EventEmitter<Producto>();
}

<article class="game-card">
  <span class="badge-discount" *ngIf="producto.precioOld">¡Oferta!</span>
  <a [routerLink]="['/producto', producto.id]">
    <img [src]="'/' + producto.img" [alt]="'Caja del juego ' + producto.nombre">
  </a>
  <div class="game-card-body">
    <h2><a [routerLink]="['/producto', producto.id]">{{ producto.nombre }}</a></h2>
    <p class="description">{{ producto.desc }}</p>
    <div class="price-row">
      <span class="price">{{ producto.precio | clp }}</span>
      <span class="price-old" *ngIf="producto.precioOld">{{ producto.precioOld | clp }}</span>
    </div>
    <span class="discount-tag discount-tag--yes" *ngIf="producto.precioOld">Con descuento</span>
    <span class="discount-tag discount-tag--none" *ngIf="!producto.precioOld">Sin descuento</span>
    <button
      type="button"
      class="add-cart"
      (click)="agregar.emit(producto)"
      [class.add-cart--ok]="feedback?.tipo === 'ok'">
      <ng-container *ngIf="feedback; else etiquetaNormal">{{ feedback.texto }}</ng-container>
      <ng-template #etiquetaNormal>Agregar al carrito</ng-template>
    </button>
  </div>
</article>
